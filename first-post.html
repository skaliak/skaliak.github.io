    New Website

    My original software portfolio (code here) ran on AngularJS. Every time it loaded, it fetched the list of my repos from the GitHub api, and grabbed the readmes of the ones that had a description set:

    var gShowcase = angular.module('gShowcase', ['ngSanitize']);

var ghusername = 'skaliak';
var baseurl = 'https://api.github.com';
var reposurl = baseurl + '/users/' + ghusername + '/repos';
var mediatype = { accept: 'application/vnd.github.VERSION.html'};  //to return md as rendered html


function gsController($scope, $http) {

    $http.get(reposurl)
        .success(function (data) {
            filtered = [];
            $scope.readmes = [];
            data.forEach(function(obj) {
                //from the list of all repos, only keep the ones with a description
               if (obj.description != null && obj.description != "")
               {
                    filtered.push(obj);
                   u = baseurl + '/repos/' + ghusername + '/' + obj.name + '/readme';
                    req_config = { headers: mediatype };
                   $http.get(u, req_config)
                       .success(function (readme_html) {
                              $scope.readmes.push(readme_html);
                       })
                       .error(function (readme_html) {
                           console.log(u);
                           console.log("error:" + readme_html.message);
                   })
               }
            });
            $scope.repos = filtered;
        })
        .error(function (data) {
            console.log("error:" + data);
        });
}

    I was pretty proud of this, but it lacked flexibility. So I decided to start over with somthing that would be more maintainable, give me more control, and allow for easy blogging. I thought about using Mezzanine, but that would require a server that could run Django, so I wouldn't be able to host it on GitHub Pages.

    I settled on using Pelican, a static site generator written in Python. I still liked the idea of generating content automatically from my GitHub readmes, so I wrote this script to fill that requirement:

    """
get all the readmes for my repos that have a description set and aren't forks
"""

import requests
import argparse

GH_API = 'https://api.github.com'
USERNAME = 'skaliak'


def get_json(theurl):
    r = requests.get(theurl)
    if r.status_code == 200:
        return r.json()


def get_filtered_repos():
    all_repos = get_json(GH_API + '/users/' + USERNAME + '/repos')
    if all_repos:
        return [x for x in all_repos if x['description'] and not x['fork']]


def get_readme_content(repo):
    url = repo['url'] + '/readme'
    r = requests.get(url)
    if r.status_code == 200:
        url = r.json()['download_url']
        r = requests.get(url)
        if r.status_code == 200:
            return r.text


def download_readme(repo, file_to_write=None, save_dir='.'):
    print "downloading readme for ", repo['name']
    content = get_readme_content(repo)
    if content:
        if file_to_write:
            file_to_write.write(content)
            file_to_write.write('\n---\n') #horizontal line
        else:
            fpath = save_dir + '/' + repo['name'] + '.md'
            with open(fpath, 'w') as f:
                f.write(content)


def download_all_readmes(save_dir='.', combine_to=None):
    repos = get_filtered_repos()
    print "found %d repos" % len(repos)
    if combine_to:
        fname = save_dir + '/' + combine_to + '.md'
        print "combining all readmes into file ", fname
        open_file = open(fname, 'w')
    else:
        open_file = None
    for repo in repos:
        download_readme(repo, open_file, save_dir)
    if open_file:
        open_file.close()


if __name__ == "__main__":
    parser = argparse.ArgumentParser(description='Download readmes from github.')
    parser.add_argument('-d', help='relative directory to download to', nargs='?', default='.')
    parser.add_argument('-c', help='file to combine into', nargs='?')

    args = parser.parse_args()

    #print "\n\ncommand would be download_all_readmes(%s, %s)\n\n" % (args.d, args.c)
    download_all_readmes(args.d, args.c)
    print "\ndone\n\n"
